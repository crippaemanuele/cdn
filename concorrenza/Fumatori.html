/**
 * IMPLEMENTAZIONE DEL PROBLEMA DEI FUMATORI
 * Ogni fumatore ha una risorsa infinita e attende le altre due.
 */

class TavoloCondiviso {
    // Semaforo Agente: 1 (disponibile all'inizio)
    Semaphore agenteSem = new Semaphore(1);
    
    // Semafori per i fumatori (inizialmente a 0)
    Semaphore f_tabacco = new Semaphore(0);
    Semaphore f_carta = new Semaphore(0);
    Semaphore f_fiammiferi = new Semaphore(0);
}

class Agente implements Runnable {
    public void run() {
        while(true) {
            tavolo.agenteSem.acquire(); // Aspetta che il tavolo sia vuoto
            int scelta = rand.nextInt(3);
            
            if(scelta == 0) {
                // Mette Carta e Fiammiferi
                tavolo.f_tabacco.release(); 
            } else if(scelta == 1) {
                // Mette Tabacco e Fiammiferi
                tavolo.f_carta.release();
            } else {
                // Mette Tabacco e Carta
                tavolo.f_fiammiferi.release();
            }
        }
    }
}

class Fumatore implements Runnable {
    public void run() {
        while(true) {
            mioSemaforo.acquire(); // Aspetta le risorse mancanti
            
            System.out.println(nome + " sta fumando...");
            Thread.sleep(1000); 
            
            contatore++; // Incrementa sigarette fumate
            tavolo.agenteSem.release(); // Libera l'agente per un nuovo ciclo
        }
    }
}
