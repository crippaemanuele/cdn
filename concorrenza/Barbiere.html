// Logica di Sincronizzazione tramite Semafori
import java.util.concurrent.Semaphore;

class BarberShop {
    private int freeSeats = 3;
    
    // Mutex per proteggere la variabile freeSeats
    private final Semaphore mutex = new Semaphore(1);
    // Segnala la presenza di clienti al barbiere
    private final Semaphore customers = new Semaphore(0);
    // Segnala che il barbiere è pronto per il cliente
    private final Semaphore barber = new Semaphore(0);

    public void getHaircut(int id) throws InterruptedException {
        mutex.acquire(); // Entra in sezione critica

        if (freeSeats > 0) {
            freeSeats--; // Occupa una sedia
            System.out.println("Cliente " + id + " si siede.");
            
            customers.release(); // Sveglia il barbiere
            mutex.release();     // Esci da sezione critica
            
            barber.acquire();    // Aspetta che il barbiere chiami
            System.out.println("Cliente " + id + " servito.");
        } else {
            mutex.release();     // Sala piena, vai via
            System.out.println("Cliente " + id + " lascia il negozio.");
        }
    }

    public void cutHair() throws InterruptedException {
        customers.acquire(); // Dorme se non ci sono clienti
        
        mutex.acquire();
        freeSeats++;         // Libera la sedia d'attesa
        barber.release();    // Pronto a servire
        mutex.release();

        // Taglio in corso...
        Thread.sleep(3000); 
    }
}
